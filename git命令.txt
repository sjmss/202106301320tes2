设置用户名和邮箱地址
git config --global user.name "username"
git config --global user.email 1111@qq.com

设置中文防止乱码
git config --global core.quotepath false 

git config --global gui.encoding utf-8

git config --global i18n.commit.encoding utf-8 

git config --global i18n.logoutputencoding utf-8 

# bash 环境下

export LESSCHARSET=utf-8

# cmd环境下：

set LESSCHARSET=utf-8
#设置环境变量LESSCHARSET=utf-8

查看用户配置
git config --list


初始化本地仓库
git init

查询暂存区和工作区状态
git status

添加文件/文件路径到暂存区
git add filename/path

移除删除暂存区的文件/文件路径
git restore --staged filename/path

移除所有暂存区的文件
git reset HEAD

查看未提交到暂存区的文件差异
git diff HEAD -- filename/path

将暂存区的更改记录提交到存储库
git commit

提交日志
git log

简化日志
git log -5 --pretty=oneline

版本回退
git reset --hard HEAD ^/~num

查询版本退回日志
git reflog

删除过的文件恢复(已经上传到本地仓库的)
git checkout -- filename/path

#查看本地仓库的文件
git ls-flies

#删除本地仓库里的文件--如果工作区有，也会同时删除工作区
git rm filename/path

分支（本地）
查看分支
git branch

新建分支并且切换到分支
git checkout -b branch01

切换分支
git checkout branch01

分支合并到主干上（在master使用）
git merge branch01

删除分支
git branch -d branch01

分支重命名大写m强制重命名
git branch -m branch02 leaf01

远程

查看本地和远程的分支
git branch -a

推送本地分支到远程
git push 远程名 本地分支名

删除远程分支(本地还在)
git push 远程仓库名 :leaf01

拉取远程分支到本地创建分支
git checkout -b leaf02 sjm2/leaf01

查询远程仓库最新状态
git fetch

查看文件内容
cat 文件名

查看冲突日志图表
git log --graph --prett=oneline
